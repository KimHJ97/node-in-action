# 비동기 에러처리

## 1. async 함수의 에러 핸들링 기본

```javascript
async function foo() {
  try {
    await new Promise((resolve, reject) => {
      setTimeout(() => {
        console.log("foo");
        reject(new Error("error입니다."));
      }, 1000);
    });
  } catch (error) {
    console.log("에러:", error);
  }
}
```

<br/>

## 2. async 함수의 에러핸들링 (Fetch API)

```javascript
async function fetchData(url) {
  try {
    const response = await fetch(url);
    if (!response.ok) {
      throw new Error(`HTTP 오류! 상태 코드: ${response.status}`);
    }
    const data = await response.json();
    console.log("데이터:", data);
  } catch (error) {
    console.error(error);
  }
}
```

<br/>

## 3. 에러위임

```javascript
async function a() {
  const res = await fetch("https://httpstat.us/404");
  if (!res.ok) throw new Error("status " + res.status);
  return await res.text();
}

async function b() {
  try {
    const result = await a();
    console.log(result);
  } catch (error) {
    console.error("에러 발생! B에서 처리", error);
  }
}
```

## 4. 리액트의 에러핸들링

```javascript
// 초기
function UserProfile() {
  const [user, setUser] = useState(null);

  useEffect(() => {
    fetch("https://jsonplaceholder.typicode.com/users/9999")
      .then((res) => res.json())
      .then((data) => setUser(data));
  }, []);

  if (!user) return null;
  return <p>안녕하세요, {user.name}님!</p>;
}

// HTTP 오류 응답은 에러로 간주하고 메시지 출력
// 응답 데이터에 name 필드가 없으면 유효하지 않은 사용자로 판단
// Fetch 자체가 실패했을 떄도 에러 메시지 노출
function UserProfile() {
  const [user, setUser] = useState(null);
  const [error, setError] = useState(null);

  useEffect(() => {
    async function loadUser() {
      try {
        const res = fetch("https://jsonplaceholder.typicode.com/users/9999");
        if (!res.ok) {
          throw new Error(`HTTP 오류: ${res.status}`);
        }
        const data = await res.json();
        if (!data.name) {
          throw new Error(`유효하지 않은 사용자입니다.`);
        }

        setUser(data);
      } catch (err) {
        setError(err.message);
      }
    }

    loadUser();
  }, []);

  if (error) return <p>❌ 에러: {error}</p>;
  if (!user) return null;
  return <p>안녕하세요, {user.name}님!</p>;
}
```

- `에러처리 위임하기`
  - 에러 상황을 캐치하지만, 에러 처리는 하지 않는다.
  - ErrorBoundary

```javascript
// 에러 던지기
function UserProfile() {
  if (error) throw error;
  // ..
}

// ErrorBoundary
function ErrorFallback({ error, resetErrorBoundary }) {
    return (
        <div role="alert">에러 발생: {error.message}</div>;
    );
}

function App() {
    return (
        <ErrorBoundary
            FallbackComponent={ErrorFallback}
            onReset={()=>{
                // reset 동작 정의 (예: 상태 초기화)
            }}
        >
            <UserProfile />
        </ErrorBoundary>
    )
}
```
